Methoden / Eigenschaften

class Car:
    _batteryCapacity = 50
    _SoC = 50

    def charge(self, power, time):
        self._SoC = self._SoC + (power * time)
    def drive(self):
        self._SoC /= 10
    @staticmethod
    def setBatteryCapacity(val):
        Car._batteryCapacity = val

Car.setBatteryCapacity(60)
tesla = Car()
tesla.drive()
tesla.charge(120, 0.5)

Vererbung

class BaseA:
	someVar = 0
	def __init__(self, someVar):
		super().__init__()
		self.someVar = someVar

class BaseB:
	pass

class Child(BaseA, BaseB):
	def __init__(self, someVar):
		super().__init__(someVar)

x = Child(5)
